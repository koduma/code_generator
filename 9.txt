#include <stdio.h>
#include <stdlib.h>
#define MAX(x,y)	((x)>(y)?(x):(y))
int main(){
    int n,m,tmp[100][100]={0},x,y,max=0;
    scanf("%d%d",&n,&m);
    for(int i=0; i<m; i++){
        scanf("%d%d",&x,&y); x--; y--;
        tmp[x][y] = 1;
        tmp[y][x] = 1;
    }
    //bit全探索
    for(int b=0; b<(1<<n); b++){
        int cnt=0, flag=1;
        for(int i=0; i<n; i++){
            if( b >> i & 1 ){
                cnt++;
                for(int j=i+1; j<n; j++){
                    if( (b >> j & 1) && !tmp[i][j] ){ flag=0; }
                }
            }
        }
        if( flag ){ max = MAX( max, cnt ); }
    }
    printf("%d\n",max);
    return 0;
}
