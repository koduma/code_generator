#include <stdio.h>

int f(int a, int b) {
  if (a > -1)
    return (b + a - 1) * (b - a) / 2;
  if (b < 2)
    return (a - b) * (a + b - 1) / 2;
  return a * (a - 1) / 2 + (b - 1) * b / 2;
}

int max(int a, int b) {
  if (a >= b)
    return a;
  else
    return b;
}

int min(int a, int b) {
  if (a <= b)
    return a;
  else
    return b;
}

int main(){
  int r, g, b, a, c, d = 1 << 30;
  scanf("%d %d %d", &r, &g, &b);
  for (int i = -300; i < 301; i++) {
    a = min(i - r, -100 - r / 2) + 100;
    c = max(i + g, 100 - b / 2) - 100;
    d = min(d, f(a, a + r) + f(i, i + g) + f(c, c + b));
  }
  printf("%d\n", d);
}
