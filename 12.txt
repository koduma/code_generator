//twosearch
#include <stdio.h>

void sort(long double array[],int left,int right)
{
	int i=left,j=right;
	long double temp;
	long double pivot;
	
	pivot=array[(left+right)/2]; // 二分区间 
	
	while(i<=j) // 
	{
		while(array[i]<pivot) 
		{
			i++;
		}
		while(array[j]>pivot)
		{
			j--;
		}
		if(i<=j)
		{
			temp=array[i];
			array[i]=array[j];
			array[j]=temp;     
			i++;
			j--;
		}
	}
	if(left<j)
	{
		sort(array,left,j);
	}
	if(i<right)
	{
		sort(array,i,right);
	}
}

int main(){
	int n,k;
	long double l[114],ans=0,t=0.5;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++)
	{
		scanf("%Lf",&l[i]);
	}
	sort(l,0,n-1);
	for(int i=0;i<k;i++)
	{
		ans+=t*l[n-1-i];
		t*=0.5;
	}
	printf("%Lf\n",ans);
	return 0;
	
}
